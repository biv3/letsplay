create or replace procedure proc_name(hire_date1 in date) is
    week_day varchar2(15);
    begin
   for c in (select to_char(hiredate,'day') as a into week_day from staff_masters where hiredate=hire_date1)
    loop
    week_day:=c.a;
    end loop;
    dbms_output.put_line(week_day);
    exception
    when no_data_found then
    dbms_output.put_line(´no data´);
    end;
    /
